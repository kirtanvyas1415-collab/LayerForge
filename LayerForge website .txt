        function selectProduct(id) {
            currentProduct = products.find(p => p.id === id);
            selectedSizeIndex = 0;
            
            document.getElementById('catalog-view').classList.add('hidden');
            document.getElementById('detail-view').classList.remove('hidden');

            document.getElementById('detail-title').innerText = currentProduct.name;
            document.getElementById('detail-desc').innerText = currentProduct.desc;

            const sizeContainer = document.getElementById('size-selector-container');
            const sizeOptionsEl = document.getElementById('size-options');
            const customContainer = document.getElementById('custom-instructions-container');
            
            if (currentProduct.id === 'custom') {
                customContainer.classList.remove('hidden');
                document.getElementById('custom-note').value = '';
            } else {
                customContainer.classList.add('hidden');
            }

            if (currentProduct.sizes.length > 1) {
                sizeContainer.classList.remove('hidden');
                sizeOptionsEl.innerHTML = currentProduct.sizes.map((s, idx) => `
                    <button onclick="selectSize(${idx})" id="size-btn-${idx}" class="size-btn px-3.5 py-2 rounded-xl border text-sm font-medium transition ${idx === 0 ? 'bg-indigo-600 border-indigo-500 text-white shadow-md' : 'bg-slate-900 border-slate-700 text-slate-300 hover:bg-slate-800'}">
                        ${s.name} ($${s.price.toFixed(2)})
                    </button>
                `).join('');
                document.getElementById('selected-size-label').innerText = currentProduct.sizes[0].name;
            } else {
                sizeContainer.classList.add('hidden');
            }

            updatePriceDisplay();
            window.scrollTo({ top: 0, behavior: 'smooth' });
        }

        function selectSize(idx) {
            selectedSizeIndex = idx;
            document.querySelectorAll('.size-btn').forEach((btn, i) => {
                if (i === idx) {
                    btn.className = 'size-btn px-3.5 py-2 rounded-xl border text-sm font-medium transition bg-indigo-600 border-indigo-500 text-white shadow-md';
                } else {
                    btn.className = 'size-btn px-3.5 py-2 rounded-xl border text-sm font-medium transition bg-slate-900 border-slate-700 text-slate-300 hover:bg-slate-800';
                }
            });
            document.getElementById('selected-size-label').innerText = currentProduct.sizes[idx].name;
            updatePriceDisplay();
        }

        function updatePriceDisplay() {
            const price = currentProduct.sizes[selectedSizeIndex].price;
            document.getElementById('detail-price').innerText = `$${price.toFixed(2)}`;
            document.getElementById('btn-price-label').innerText = `$${price.toFixed(2)}`;
        }

        function setColor(name, colorClass) {
            selectedColor = name;
            document.getElementById('selected-color-name').innerText = name;
            document.querySelectorAll('.color-btn').forEach(btn => {
                if (btn.getAttribute('data-name') === name) {
                    btn.classList.add('border-white', 'ring-2', 'ring-indigo-500/60');
                } else {
                    btn.classList.remove('border-white', 'ring-2', 'ring-indigo-500/60');
                    btn.classList.add('border-transparent');
                }
            });
        }

        function showCatalog() {
            document.getElementById('detail-view').classList.add('hidden');
            document.getElementById('catalog-view').classList.remove('hidden');
            window.scrollTo({ top: 0, behavior: 'smooth' });
        }

        function addCurrentProductToCart() {
            const sizeObj = currentProduct.sizes[selectedSizeIndex];
            let customNote = '';

            if (currentProduct.id === 'custom') {
                customNote = document.getElementById('custom-note').value.trim();
                if (!customNote) {
                    alert('Please enter a description for your custom print request!');
                    return;
                }
            }

            cart.push({
                name: currentProduct.name,
                size: sizeObj.name,
                price: sizeObj.price,
                color: selectedColor,
                customNote: customNote
            });

            updateCartBadge();
            
            const btn = document.querySelector('#detail-view button:last-of-type');
            const originalHTML = btn.innerHTML;
            btn.innerHTML = `✓ Added to Cart!`;
            btn.classList.remove('bg-indigo-600', 'hover:bg-indigo-500');
            btn.classList.add('bg-emerald-600');

            setTimeout(() => {
                btn.innerHTML = originalHTML;
                btn.classList.remove('bg-emerald-600');
                btn.classList.add('bg-indigo-600', 'hover:bg-indigo-500');
            }, 1200);
        }

        function updateCartBadge() {
            const badge = document.getElementById('cart-badge');
            if (cart.length > 0) {
                badge.innerText = cart.length;
                badge.classList.remove('hidden');
            } else {
                badge.classList.add('hidden');
            }
        }

        function openCartModal() {
            const container = document.getElementById('cart-items-container');
            const totalEl = document.getElementById('cart-total-price');

            if (cart.length === 0) {
                container.innerHTML = `<div class="text-center py-8 text-slate-500 text-sm">Your cart is currently empty.</div>`;
                totalEl.innerText = '$0.00';
            } else {
                let total = 0;
                container.innerHTML = cart.map((item, index) => {
                    total += item.price;
                    return `
                        <div class="bg-slate-950/60 border border-slate-800 rounded-2xl p-3 flex justify-between items-start text-sm">
                            <div>
                                <h4 class="font-bold text-white">${item.name}</h4>
                                <p class="text-xs text-slate-400">${item.size} &bull; ${item.color}</p>
                                ${item.customNote ? `<p class="text-xs text-indigo-300 mt-1 italic bg-indigo-950/40 p-1.5 rounded border border-indigo-500/20">"${item.customNote}"</p>` : ''}
                            </div>
                            <div class="flex items-center gap-3">
                                <span class="font-semibold text-indigo-400">$${item.price.toFixed(2)}</span>
                                <button onclick="removeFromCart(${index})" class="text-slate-500 hover:text-red-400 transition">
                                    <svg xmlns="http://www.w3.org/2000/svg" class="h-4 w-4" fill="none" viewBox="0 0 24 24" stroke="currentColor">
                                        <path stroke-linecap="round" stroke-linejoin="round" stroke-width="2" d="M19 7l-.867 12.142A2 2 0 0116.138 21H7.862a2 2 0 01-1.995-1.858L5 7m5 4v6m4-6v6m1-10V4a1 1 0 00-1-1h-4a1 1 0 00-1 1v3M4 7h16" />
                                    </svg>
                                </button>
                            </div>
                        </div>
                    `;
                }).join('');
                totalEl.innerText = `$${total.toFixed(2)}`;
            }

            document.getElementById('cart-modal').classList.remove('hidden');
        }

        function removeFromCart(index) {
            cart.splice(index, 1);
            updateCartBadge();
            openCartModal();
        }

        function closeCartModal() {
            document.getElementById('cart-modal').classList.add('hidden');
        }

        function checkoutOrder() {
            if (cart.length === 0) return;
            const total = cart.reduce((sum, item) => sum + item.price, 0);
            alert(`Thank you for ordering with LayerForge! Total amount: $${total.toFixed(2)}. Preparing your custom 3D prints for production!`);
            cart = [];
            updateCartBadge();
            closeCartModal();
            showCatalog();
        }
